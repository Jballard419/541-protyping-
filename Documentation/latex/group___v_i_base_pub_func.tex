\begin{DoxyParams}[1]{Parameters}

\begin{DoxyReturn}{Returns}
A float between 0.\+0 and 1.\+0 where 0.\+0 means that the waveform should be made silent and 1.\+0 means that it should be unaltered.
\end{DoxyReturn}
