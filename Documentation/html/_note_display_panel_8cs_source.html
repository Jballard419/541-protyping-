222  /**
223  * @brief Handles when the @link NoteDisplayPanel::mRemoveNoteButton remove button@endlink is clicked.
224  */
225  public void OnRemoveButtonClicked()
226  {
227  // Set the note index for all notes after this one
228  int sib = transform.GetSiblingIndex() + 1;
229  int index = mNoteIndex;
230  while( sib < transform.parent.childCount )
231  {
232  if( transform.parent.GetChild( sib ).GetComponent<SC_NoteDisplayPanel>() != null )
233  {
234  transform.parent.GetChild( sib ).GetComponent<SC_NoteDisplayPanel>().SetNoteIndex( index );
235  index++;
236  }
237  sib++;
238  }
239 
240  // Signal that this note display should be removed.
241  mParent.RemoveNote( this );
242 
243  // Signal that this note should be removed from the song.
244  GameObject.Find( "SongCreationCanvas" ).GetComponent<SongCreation>().OnRemoveNote( mNoteIndex );
245  }

258
