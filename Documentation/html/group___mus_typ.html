Detailed Description

Allows for testing the audio with the computer keyboard.

Information

Musical Typing allows for using a computer keyboard to debug sound output by triggering note events for a 19-note range whenever specific keys on the keyboard are pressed/released. The velocities for each key can be set with DEBUG_SetMusicalTypingKeyVelocity(). The keys that are used are (in this order) a, w, s, e, d, f, t, g, y, h, u, j, k, o, l, p, ;, ', and ]

185  /**
186  * @brief Handler for Musical Typing Events
187  * @param[in] aKeyEvent A GUI keyboard event triggered by a key being pressed or released.
188  *
189  * This function sees if a pressed/released key is one that is used for Musical Typing.
190  * If so, it simulates a note event based on the pressed/released key.
191  */
192  private void OnMusicalTypingEvent( Event aKeyEvent )
193  {
194  // Check if a musical typing key is pressed or released and fire off the
195  // corresponding PlayNote or NoteFadeOut event if so. The debug velocity is
196  // used for the events.
197  int i = 0;
198  int numActive = mVIM.GetNumActiveNotes();
199  while( MusicalTypingEnabled && i < DEBUG_numMusicalTypingKeys && i < numActive )
200  {
201  // See if the key is a Musical Typing key.
202  if( aKeyEvent.keyCode == DEBUG_musicalTypingKeys[i] )
203  {
204  // If the key was pressed, then simulate a NotePlayEvent.
205  if( aKeyEvent.type == EventType.KeyDown && !mPressedKeys[i] )
206  {
207  #if DEBUG_AUDIO_DIAGNOSTICS
208  mDiagnosticsHandler.SetInputTime.Invoke();
209  #endif
210 
211  // Update the pressed keys
212  mPressedKeys[i] = true;
213 
214  // Get the velocity that should be used.
215  int velocity = mKeyVelocities[i];
216  if( RandomizeVelocities )
217  {
218  velocity = UnityEngine.Random.Range( mRandomVelocityRange[0], mRandomVelocityRange[1] );
219  }
220  mVIM.PlayNote.Invoke( mVIM.GetActiveNotes()[i], velocity );
221  }
222  // If the key was released, then simulate a ReleaseNoteEvent.
223  if( aKeyEvent.type == EventType.KeyUp )
224  {
225  // Update the pressed keys
226  mPressedKeys[i] = false;
227 
228  mVIM.ReleaseNote.Invoke( mVIM.GetActiveNotes()[i] );
229  }
230  }
231  i++;
232  }
233  }
