Detailed Description

A C# class that represents a song.

/**

Information

This class's responsibilities include:

  • Serving as a container for a pre-arranged series of notes.
  • Converting the abstract representions of musical notes into a form that relates to the actual waveforms of the audio.
  • Saving and loading copies of itself to/from a file.

149  /**
150  * @brief Adds a @link Music::CombinedNote note@endlink to the Song.
151  * @param[in] aNewNote The @link Music::CombinedNote note@endlink to add.
152  */
153  public void AddNote( Music.CombinedNote aNewNote )
154  {
155  // Update the number of musical notes in the song if needed.
156  if( aNewNote.MusicalNote.Pitches != null )
157  {
158  mNumMusicalNotes++;
159 
160  for( int i = 0; i < aNewNote.MusicalNote.Pitches.Length; i++ )
161  {
162  // Update the highest pitch.
163  if( aNewNote.MusicalNote.Pitches[i] != Music.PITCH.REST && mHighestPitch < aNewNote.MusicalNote.Pitches[i] )
164  {
165  mHighestPitch = aNewNote.MusicalNote.Pitches[i];
166  }
167 
168  // Update the lowest pitch.
169  if( aNewNote.MusicalNote.Pitches[i] != Music.PITCH.REST && mLowestPitch > aNewNote.MusicalNote.Pitches[i] )
170  {
171  mLowestPitch = aNewNote.MusicalNote.Pitches[i];
172  }
173  }
174 
175  }
176 
177  // Update the number of notes with drum hits in the song if needed.
178  if( aNewNote.Drums.Hits != null )
179  {
180  mNumDrumNotes++;
181  }
182 
183  // Add the note to the list.
184  mNotes.Add( aNewNote );
185 
186  // Update the song type.
187  UpdateSongType();
188  }

243  /**
244  * @brief Converts the @link Music::CombinedNote notes@endlink in the song to NoteData.
245  * @return An array of @link Song::CombinedNoteData note data@endlink representing the @link Music::CombinedNote notes@endlink in the Song.
246  *
247  * This function is used to convert the Song's @link Music::CombinedNote abstract information@endlink
248  * which is based around musical notation into @link Song::CombinedNoteData specific information@endlink
249  * that can be used to map it to audio data and waveform samples.
250  *
251  * @see VirtualInstrumentManager::OnPlaySongEvent VirtualInstrumentManager::OnPlayDrumLoopEvent NoteOutputObject::OnAudioFilterRead
252  */
253  public CombinedNoteData[] GetNoteData()
254  {
255  // Allocate the returned array.
256  CombinedNoteData[] returned = new CombinedNoteData[mNotes.Count];
257 
258  // Set up variables to keep track of a note's number of samples and total offset.
259  int offset = 0;
260  int numSamp = 0;
261 
262  // Iterate through each note in the list.
263  int index = 0;
264  foreach( Music.CombinedNote note in mNotes )
265  {
266  // Get the number of samples and the total offset for the note.
267  numSamp = GetNoteLengthInSamples( mBPM, 44100, note.MusicalNote.Length, mTimeSignature );
268  offset += GetNoteLengthInSamples( mBPM, 44100, note.OffsetFromPrevNote, mTimeSignature );
269 
270  // Set the values for the NoteData's MelodyData.
271  returned[index].MelodyData.NumSamples = numSamp;
272  returned[index].MelodyData.Pitches = note.MusicalNote.Pitches;
273  returned[index].MelodyData.Velocity = note.MusicalNote.Velocity;
274 
275  // Set the values for the NoteData's PercussionData.
276  returned[index].PercussionData.Velocity = note.Drums.Velocity;
277  returned[index].PercussionData.Hits = note.Drums.Hits;
278  returned[index].PercussionData.HasHiHat = note.Drums.HasHiHat;
279 
280  // Set the offset for the NoteData.
281  returned[index].TotalOffset = offset;
282 
283  // Go to the next note.
284  index++;
285  }
286 
287  return returned;
288  }

341  // Update information about the pitches in the song..
342  if( removedNote.MusicalNote.Pitches != null )
343  {
344  // Decrease the number of musical notes.
345  mNumMusicalNotes--;
346 
347  // See if we need to set a new highest/lowest pitch.
348  foreach( Music.PITCH pitch in removedNote.MusicalNote.Pitches )

475  /**
476  * @brief Writes the song to a file. The file will be stored in StreamingAssets/Songs/.
477  * @see @link DocSongFileFormat Song File Format@endlink
478  */
479  public void WriteSongToFile()
480  {
481  // Set up the filepath.
482  string filepath = SONG_FILE_PATH;
483  switch( mType )
484  {
485  case SongType.DrumLoop:
486  filepath += "DRUMLOOP_" + mName + ".song";
487  break;
488  case SongType.Melody:
489  filepath += "MELODY_" + mName + ".song";
490  break;
491  case SongType.CombinedMelodyAndPercussion:
492  filepath += "SONG_" + mName + ".song";
493  break;
494  default:
495  break;
496  }
497 
498  // Set an array for the contents of the file
499  string[] contents = new string[mNotes.Count + 2];
500 
501  // Put the name of the song in first line to be saved.
502  contents[0] = mName;
503 
504  // Put the song type, default BPM and the time signature values in the second line to be saved.
505  contents[1] = ( (int)mType ).ToString() + ";" + mBPM.ToString() + ";" +
506  mTimeSignature.BeatsPerMeasure.ToString() + ";" + ( (int)mTimeSignature.BaseBeat ).ToString();
507 
508  // Put all the notes into the lines from line 3 to line [numNotes].
509  for( int i = 0; i < mNotes.Count; i++ )
510  {
511  // Set up the note string.
512  contents[i + 2] = "";
513 
514  // If pitches exist for the song, then add them.
515  if( mType != SongType.DrumLoop )
516  {
517  // Add the pitches if they exist for this note.
518  if( mNotes[i].MusicalNote.Pitches != null )
519  {
520  // Add the pitches to the string
521  for( int j = 0; j < mNotes[i].MusicalNote.Pitches.Length; j++ )
522  {
523  contents[i + 2] += ( (int)mNotes[i].MusicalNote.Pitches[j] ).ToString();
524 
525  // If this is not the last pitch of the note, then add a comma
526  if( j + 1 != mNotes[i].MusicalNote.Pitches.Length )
527  {
528  contents[i + 2] += ",";
529  }
530  // If this is the last pitch of the note, then add a bar or a semicolon depending on if there are drums.
531  else
532  {
533  if( mType == SongType.CombinedMelodyAndPercussion )
534  {
535  contents[i + 2] += "|";
536  }
537  else
538  {
539  contents[i + 2] += ";";
540  }
541  }
542  }
543  }
544  // If pitches don't exist for the note, then put null.
545  else
546  {
547  contents[i + 2] += "null|";
548  }
549  }
550 
551  // If drums exist for the song, then add them.
552  if( mType != SongType.Melody )
553  {
554  // If there are drums for this note, then add them.
555  if( mNotes[i].Drums.Hits != null )
556  {
557  // Add the drums to the string
558  for( int j = 0; j < mNotes[i].Drums.Hits.Length; j++ )
559  {
560  contents[i + 2] += ( (int)mNotes[i].Drums.Hits[j] ).ToString();
561 
562  // If this is not the last drum hit of the note, then add a comma
563  if( j + 1 != mNotes[i].Drums.Hits.Length )
564  {
565  contents[i + 2] += ",";
566  }
567  // If this is the last pitch of the note, then add a semicolon.
568  else
569  {
570  contents[i + 2] += ";";
571  }
572  }
573  }
574  }
575 
576  // If the song is not a drum loop, then add the note length.
577  if( mType != SongType.DrumLoop )
578  {
579  contents[i + 2] += ( ( (int)mNotes[i].MusicalNote.Length ).ToString() + ";" );
580  }
581 
582  // Add the offset from the previous note.
583  contents[i + 2] += ( ( (int)mNotes[i].OffsetFromPrevNote ).ToString() + ";" );
584 
585  // If the song is not a drum loop, then add the musical note velocity.
586  if( mType != SongType.DrumLoop )
587  {
588  contents[i + 2] += mNotes[i].MusicalNote.Velocity.ToString();
589  }
590 
591  // Put a bar if both drums and pitches exist for the song.
592  if( mType == SongType.CombinedMelodyAndPercussion )
593  {
594  contents[i + 2] += "|";
595  }
596 
597  // If the song has drums, then put the velocity for the drums for this note.
598  if( mType != SongType.Melody )
599  {
600  contents[i + 2] += mNotes[i].Drums.Velocity;
601  }
602 
603  }
604 
605  // Write the contents to the file.
606  System.IO.File.WriteAllLines( filepath, contents );
607  }

632  // Open the file stream.
633  StreamReader parser = new StreamReader( aSongFilePath );
634 
635  // Get the name of the song
636  string curLine = parser.ReadLine();
637  mName = curLine;
638 
639  // Get the song type.
640  curLine = parser.ReadLine();
641  string[] splitLine = curLine.Split( ';' );
642  mType = (SongType)int.Parse( splitLine[0] );
643 
644  // Get the default BPM
645  mBPM = int.Parse( splitLine[1] );
646 
647  // Get the time signature
648  mTimeSignature.BeatsPerMeasure = int.Parse( splitLine[2] );
649  mTimeSignature.BaseBeat = (Music.NOTE_LENGTH)int.Parse( splitLine[3] );
650 
651  // Get the notes of the song.
652  curLine = parser.ReadLine();
653  while( curLine != null )
654  {
655  // Get the line and reset the split line index.
656  splitLine = curLine.Split( ';' );
657  splitLineIndex = 0;
658 
659  // Get the pitches for the note if needed.
660  if( mType != SongType.DrumLoop )
661  {
662  curNotePitches = ParsePitches( splitLine[splitLineIndex] );
663  }
664 
665  // Get the drums for the note if needed.
666  if( mType != SongType.Melody )
667  {
668  curNoteDrums = ParseDrums( splitLine[splitLineIndex] );
669  }
670 
671  // Go to the next section of the line.
672  splitLineIndex++;
673 
674  // If needed, get the melody note length.
675  if( mType != SongType.DrumLoop )
676  {
677  curNoteLength = (Music.NOTE_LENGTH)int.Parse( splitLine[splitLineIndex] );
678  splitLineIndex++;
679  }
680 
681  // Get the offset from the previous note.
682  Music.NOTE_LENGTH offset = (Music.NOTE_LENGTH)int.Parse( splitLine[splitLineIndex] );
683  splitLineIndex++;
684 
685  // Get the velocity/velocities of the note.
686  if( mType == SongType.CombinedMelodyAndPercussion )
687  {
688  string[] velocityString = splitLine[splitLineIndex].Split( '|' );
689  curNoteMelodyVelocity = int.Parse( velocityString[0] );
690  curNoteDrumVelocity = int.Parse( velocityString[1] );
691  }
692  else if( mType == SongType.Melody )
693  {
694  curNoteMelodyVelocity = int.Parse( splitLine[splitLineIndex] );
695  }
696  else
697  {
698  curNoteDrumVelocity = int.Parse( splitLine[splitLineIndex] );
699  }
700 
701  // Add the note.
702  AddNote( Music.CreateNote( curNoteMelodyVelocity, curNoteLength, curNotePitches, curNoteDrumVelocity, curNoteDrums, offset ) );
703 
704  // Get the next line
705  curLine = parser.ReadLine();
706  }
707  }

864  /**
865  * @brief Gets the number of waveform samples required to accurately represent the @link Music::NOTE_LENGTH length of a note@endlink for a given @link DefBPM BPM@endlink and @link Music::TimeSignature time signature@endlink.
866  * @param[in] aBPM The @link DefBPM BPM@endlink that should be accounted for.
867  * @param[in] aSampleRate The sample rate of the note. Might need to remove since most likely everything's going to have a 44\.1KHz sample rate.
868  * @param[in] aNoteLength The @link Music::NOTE_LENGTH note length@endlink that's being converted into a number of waveform samples.
869  * @param[in] aTimeSignature The @link Music::TimeSignature time signature@endlink that should be accounted for.
870  * @return The number of waveform samples that represents the given @link Music::NOTE_LENGTH note length@endlink at the given @link DefBPM BPM@endlink and @link Music::TimeSignature time signature@endlink.
871  *
872  */
873  public static int GetNoteLengthInSamples( int aBPM, int aSampleRate, Music.NOTE_LENGTH aNoteLength, Music.TimeSignature aTimeSignature )
874  {
875  // Initialize variables for calculating the note length.
876  float beatsPerSecond = (float)aBPM / 60f;
877 
878  // Since the audio data is split into two channels with even indices for the data
879  // in the left channel and odd indices for the data in the right channel, the
880  // actual conversion for number of samples per second is 2 * sample rate.
881  float numSamplesPerBeat = 2f * ( 1f / beatsPerSecond ) * (float)aSampleRate;
882 
883  // Calculate the length for various time signatures by relating the base beat to
884  // a quarter note.
885  float numQuarterNotesPerBeat = 1f;
886 
887  // Right now, only provide support for base beats of 4 and 8.
888  switch( aTimeSignature.BaseBeat )
889  {
890  case Music.NOTE_LENGTH.E:
891  numQuarterNotesPerBeat /= 2f;
892  break;
893  default:
894  break;
895  }
896 
897  // Calculate the note length in samples.
898  switch( aNoteLength )
899  {
900  case Music.NOTE_LENGTH.T:
901  return (int)( numSamplesPerBeat / ( 8f * numQuarterNotesPerBeat ) );
902  case Music.NOTE_LENGTH.D_T:
903  return (int)( 1.5f * numSamplesPerBeat / ( 8f * numQuarterNotesPerBeat ) );
904  case Music.NOTE_LENGTH.S:
905  return (int)( numSamplesPerBeat / ( 4f * numQuarterNotesPerBeat ) );
906  case Music.NOTE_LENGTH.D_S:
907  return (int)( 1.5f * numSamplesPerBeat / ( 4f * numQuarterNotesPerBeat ) );
908  case Music.NOTE_LENGTH.E:
909  return (int)( numSamplesPerBeat / ( 2f * numQuarterNotesPerBeat ) );
910  case Music.NOTE_LENGTH.D_E:
911  return (int)( 1.5f * numSamplesPerBeat / ( 2f * numQuarterNotesPerBeat ) );
912  case Music.NOTE_LENGTH.Q:
913  return (int)( numSamplesPerBeat / numQuarterNotesPerBeat );
914  case Music.NOTE_LENGTH.D_Q:
915  return (int)( numSamplesPerBeat * ( 1.5f / numQuarterNotesPerBeat ) );
916  case Music.NOTE_LENGTH.H:
917  return (int)( numSamplesPerBeat * ( 2f / numQuarterNotesPerBeat ) );
918  case Music.NOTE_LENGTH.D_H:
919  return (int)( numSamplesPerBeat * ( 3f / numQuarterNotesPerBeat ) );
920  case Music.NOTE_LENGTH.W:
921  return (int)( numSamplesPerBeat * ( 4f / numQuarterNotesPerBeat ) );
922  case Music.NOTE_LENGTH.D_W:
923  return (int)( numSamplesPerBeat * ( 6f / numQuarterNotesPerBeat ) );
924  default:
925  return 0;
926  }
927  }
