50  // Adds a song to the SongManager
51  // IN: aSong A new song.
52  public void AddSong( Song aSong )
53  {
54  // Sanity Check
55  Assert.IsNotNull( aSong, "Tried to add a song that didn't exist!" );
56 
57  // Create the song entry.
58  switch( aSong.GetSongType() )
59  {
60  case Song.SongType.Melody:
61  mMelodies.Add( aSong.GetName(), aSong );
62  break;
63  case Song.SongType.DrumLoop:
64  mDrumLoops.Add( aSong.GetName(), aSong );
65  mNumDrumLoops++;
66  break;
67  case Song.SongType.CombinedMelodyAndPercussion:
68  mCombinedSongs.Add( aSong.GetName(), aSong );
69  mNumCombinedSongs++;
70  break;
71  }
72  }

135  else if( mDrumLoops.ContainsKey( aSongName ) )
136  {
137  return mDrumLoops[aSongName];

198  // Loads songs in the folder "APP_PATH/Resources/Audio/Songs/".
199  // Songs have the format "SONG_[songname].song
200  public void LoadSongs()
201  {
202  // Get the info for the files in the folder.
203  DirectoryInfo info = new DirectoryInfo( Song.SONG_FILE_PATH );
204 
205  // Iterate through each file in the folder and load a new song if the filename matches
206  // the format.
207  foreach( FileInfo file in info.GetFiles() )
208  {
209  if( file.Name.EndsWith( ".song" ) )
210  {
211  // Load the new song.
212  Song newSong = new Song( file.FullName );
213 
214  // Add the new song to the proper container.
215  if( file.Name.StartsWith( "SONG_" ) )
216  {
217  mCombinedSongs.Add( newSong.GetName(), newSong );
218  mNumCombinedSongs++;
219  }
220  else if( file.Name.StartsWith( "MELODY_" ) )
221  {
222  mMelodies.Add( newSong.GetName(), newSong );
223  mNumMelodies++;
224  }
225  else if( file.Name.StartsWith( "DRUMLOOP" ) )
226  {
227  mDrumLoops.Add( newSong.GetName(), newSong );
228  mNumDrumLoops++;
229  }
230  }
231 
232  }
