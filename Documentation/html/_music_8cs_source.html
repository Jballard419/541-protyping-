37  /*************************************************************************//**
38  * @defgroup MusicEnums Enums
39  * @ingroup DocMusic
40  * These are enums which help other classes organize anything related to music.
41  * Enums are pretty much just integer arrays with unique identifiers instead of indices. They're used to help make code more readable.
42  * An example of using an enum would be: @code Music.PITCH newNotePitch = Music.PITCH.C4; @endcode
43  * The actual values of each member of these enums (remember that it's pretty much an array of integers)
44  * range from 0 to the number of members minus 1, and these values can be retrieved by typecasting.
45  * For example:
46  * @code int noteIndex = (int)Music.PITCH.C4; // noteIndex now holds the value 48. @endcode
47  * noteIndex holds the value 48 because @link Music::PITCH::C4 C4@endlink is the 49th member of the @link Music::PITCH pitch enum@endlink which starts from 0.
48  * The members values can be typecast to int,float,... and vice-versa.
49  * For example:
50  * @code Music.PITCH newPitch = (Music.PITCH)48; // newPitch now holds the value Music.PITCH.C4 @endcode

315  /**
316  * @brief Creates a new CombinedNote struct from the parameters.
317  * @param[in] aMelodyVelocity The velocity of the MelodyNote
318  * @param[in] aLength The length of the MelodyNote
319  * @param[in] aPitches The pitches of the MelodyNote
320  * @param[in] aDrumVelocity The velocity of the PercussionNote
321  * @param[in] aDrumHits The hits of the PercussionNote
322  * @param[in] aOffsetFromPrevNote The offset from the previous CombinedNote.
323  */
324  public static CombinedNote CreateNote( int aMelodyVelocity, NOTE_LENGTH aLength, PITCH[] aPitches, int aDrumVelocity, DRUM[] aDrumHits, NOTE_LENGTH aOffsetFromPrevNote )
325  {
326  // Create the new note struct.
327  CombinedNote newNote = new CombinedNote();
328 
329  // Set up the melody note.
330  newNote.MusicalNote.Velocity = aMelodyVelocity;
331  newNote.MusicalNote.Length = aLength;
332  newNote.MusicalNote.Pitches = aPitches;
333 
334  // Set up the percussion note.
335  newNote.Drums.Velocity = aDrumVelocity;
336  newNote.Drums.Hits = aDrumHits;
337  newNote.Drums.HasHiHat = false;
338  if( aDrumHits != null )
339  {
340  // Check if the new note has any hi hat hits.
341  newNote.Drums.HasHiHat = false;
342  for( int i = 0; i < aDrumHits.Length; i++ )
343  {
344  if( aDrumHits[i] == Music.DRUM.HIHAT_C ||
345  aDrumHits[i] == Music.DRUM.HIHAT_O ||
346  aDrumHits[i] == Music.DRUM.HIHAT_P )
347  {
348  newNote.Drums.HasHiHat = true;
349  }
350  }
351  }
352 
353  // Set the offset.
354  newNote.OffsetFromPrevNote = aOffsetFromPrevNote;
355 
356  // Return the new note.
357  return newNote;
358  }

382  /**
383  * @brief Gets the percentage that a note length takes up in a measure for the given time signature.
384  * @param[in] aLength the given NOTE_LENGTH that represents the length of a note.
385  * @param[in] aTimeSignature The given TimeSignature
386  */
387  public static float GetNoteLengthRelativeToMeasure( NOTE_LENGTH aLength, TimeSignature aTimeSignature )
388  {
389  // Relate everything to quarter notes.
390  float quarterNoteLength = 1f;
391 
392  // Get the ratio of quarter note to beat.
393  switch( aTimeSignature.BaseBeat )
394  {
395  case NOTE_LENGTH.E:
396  quarterNoteLength *= 2f;
397  break;
398  default:
399  break;
400 
401  }
402 
403  // Get the percentage that a quarter note takes up in the measure.
404  quarterNoteLength /= (float)aTimeSignature.BeatsPerMeasure;
405 
406  // Return the percentage that the given note length takes up in a measure.
407  switch( aLength )
408  {
409  case Music.NOTE_LENGTH.T:
410  return quarterNoteLength / 8f;
411  case Music.NOTE_LENGTH.D_T:
412  return 1.5f * quarterNoteLength / 8f;
413  case Music.NOTE_LENGTH.S:
414  return quarterNoteLength / 4f;
415  case Music.NOTE_LENGTH.D_S:
416  return 1.5f * quarterNoteLength / 4f;
417  case Music.NOTE_LENGTH.E:
418  return quarterNoteLength / 2f;
419  case Music.NOTE_LENGTH.D_E:
420  return 1.5f * quarterNoteLength / 2f;
421  case Music.NOTE_LENGTH.Q:
422  return quarterNoteLength;
423  case Music.NOTE_LENGTH.D_Q:
424  return quarterNoteLength * 1.5f;
425  case Music.NOTE_LENGTH.H:
426  return quarterNoteLength * 2f;
427  case Music.NOTE_LENGTH.D_H:
428  return quarterNoteLength * 3f;
429  case Music.NOTE_LENGTH.W:
430  return quarterNoteLength * 4f;
431  case Music.NOTE_LENGTH.D_W:
432  return quarterNoteLength * 6f;
433  default:
434  return 0;
435  }
436  }
