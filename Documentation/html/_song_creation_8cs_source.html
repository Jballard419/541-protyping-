634  */
635  public void OnCreateNote()
636  {
637  // Sanity check.
638  if( mEditing )
639  {
640  Assert.IsFalse( !mEditing, "Tried to create a note in the song creation interface when we should have been editing one!" );
641  return;
642  }
643 
644  // Get the selected pitches
645  Music.PITCH[] pitches = mPitchSelector.GetSelectedPitches();
646 
647  // Get the selected drums.
648  Music.PITCH[] drums = mDrumSelector.GetSelectedPitches();
649 
650  // Make sure that some pitches or drums are actually selected.
651  if( pitches != null || drums != null )
652  {
653  // Get the offset and velocity.
654  Music.NOTE_LENGTH length = Music.NOTE_LENGTH.NONE;
655  mLastLength = mLengthPanel.GetSelected();
656  Music.NOTE_LENGTH offset = mOffsetPanel.GetSelected();
657  int pitchVelocity = 0;
658  int drumVelocity = 0;
659 
660  // If the song doesn't have any notes, then set the offset of the first note to none.
661  if( mSong.GetNumNotes() == 0 )
662  {
663  offset = Music.NOTE_LENGTH.NONE;
664  }
665 
666  // Convert the selected pitches from the drum selector into actual drum hits.
667  Music.DRUM[] Hits = null;
668  if( drums != null )
669  {
670  Hits = new Music.DRUM[drums.Length];
671  for( int i = 0; i < drums.Length; i++ )
672  {
673  Hits[i] = (Music.DRUM)drums[i];
674  }
675  }
676 
677 
678  // Update the velocity/length for the pitches and drums if needed.
679  if( pitches != null )
680  {
681  pitchVelocity = (int)mVelocitySlider.value;
682  length = mLengthPanel.GetSelected();
683  }
684 
685  if( drums != null )
686  {
687  drumVelocity = (int)mVelocitySlider.value;
688  }
689 
690 
691  // Add the note to the song.
692  mSong.AddNote( Music.CreateNote( pitchVelocity, length, pitches, drumVelocity, Hits, offset ) );

739

741  mPitchSelector.SetPitches( noteBeingModified.MusicalNote.Pitches );
742 
743  // Set the drum selector to have the drums from the note.
744  Music.PITCH[] drumsAsPitches = null;

752  index++;
753  }
754  }
755  mDrumSelector.SetPitches( drumsAsPitches );
756  mLengthPanel.SetSelected( noteBeingModified.MusicalNote.Length );
757  mOffsetPanel.SetSelected( noteBeingModified.OffsetFromPrevNote );
758  mVelocitySlider.value = Mathf.Max( noteBeingModified.Drums.Velocity, noteBeingModified.MusicalNote.Velocity );
759  }
760  // Handle the case where we're editing a note/note panel and it calls this
761  // event while it's being edited. This means that the edit should be cancelled.
762  else
763  {
764  // Change the new note button back to its original state.
765  mNewNoteButton.transform.GetChild( 0 ).GetComponent<Text>().text = "Insert Note";
766  mNewNoteButton.onClick.RemoveListener( OnModifyNote );
767  mNewNoteButton.onClick.AddListener( OnCreateNote );

770  mEditPanel.StopEditing();
771

774  mOffsetPanel.SetSelected( Music.NOTE_LENGTH.NONE );

823  // Modifies a note in the song.
824  public void OnModifyNote()
825  {
826  // Make sure that we're actually editing a note.
827  if( mEditing )
828  {
829  // Sanity check.
830  if( mEditPanel == null )
831  {
832  Assert.IsNotNull( mEditPanel, "Tried to modify a note, but the note panel in the song creator was null!" );
833  return;
834  }
835 
836  // Get the pitches
837  Music.PITCH[] pitches = mPitchSelector.GetSelectedPitches();
838 
839  // Get the drums
840  Music.PITCH[] drums = mDrumSelector.GetSelectedPitches();
841 
842  // Make sure that some pitches or drums are actually selected.
843  if( pitches != null || drums != null )
844  {
845  // Get the index of the note in the song and the images for the length/offset.
846  int index = mEditPanel.GetNoteIndex();
847  Sprite[] images = mNoteDisplay.GetSprites();
848 
849  // Convert the selected pitches from the drum selector into actual drum hits.
850  Music.DRUM[] Hits = null;
851  if( drums != null )
852  {
853  Hits = new Music.DRUM[drums.Length];
854  for( int i = 0; i < drums.Length; i++ )
855  {
856  Hits[i] = (Music.DRUM)drums[i];
857  }
858  }
859 
860  // Set the values of the note struct.
861  Music.NOTE_LENGTH length = Music.NOTE_LENGTH.NONE;
862  Music.NOTE_LENGTH offset = mOffsetPanel.GetSelected();
863  int pitchVelocity = 0;
864  int drumVelocity = 0;
865 
866  // Update the velocity for the pitches and drums if needed.
867  if( pitches != null )
868  {
869  pitchVelocity = (int)mVelocitySlider.value;
870  length = mLengthPanel.GetSelected();
871  }
872 
873  if( drums != null )
874  {
875  drumVelocity = (int)mVelocitySlider.value;
876  }
877 
878  // Replace the note in the song with the modified note.
879  mSong.ReplaceNote( Music.CreateNote( pitchVelocity, length, pitches, drumVelocity, Hits, offset ), index );
880 
881  // Get the string of pitches for the modified note.
882  string pitchString = "None";
883  if( pitches != null )
884  {
885  pitchString = "";
886  for( int i = 0; i < pitches.Length; i++ )
887  {
888  if( pitches[i] == Music.PITCH.REST )
889  {
890  pitchString += "Rest ";
891  }
892  else
893  {
894  pitchString += ( Music.NoteToString( pitches[i] ) + " " );
895  }
896  }
897  }
898 
899  // Get the string of drums for the modified note.
900  string drumString = "None";
901  if( Hits != null )
902  {
903  drumString = "";
904  foreach( Music.DRUM drum in Hits )
905  {
906  drumString += Music.DrumToString( drum ) + " ";
907  }
908  }
909 
910  // Update the panel representing the note.
911  mEditPanel.SetDrums( drumString );
912  mEditPanel.SetPitches( pitchString );
913  mEditPanel.SetLengthImage( images[(int)length] );
914  mEditPanel.SetOffsetImage( images[(int)offset] );
915  mEditPanel.SetOffset( offset );
916  mEditPanel.SetVelocity( mVelocitySlider.value.ToString() );
917 
918  // Mark that the panel is no longer being edited.
919  mEditPanel.StopEditing();
920 
921  // Update the new note button so that it shows that we're no longer editing a note.
922  mNewNoteButton.transform.GetChild( 0 ).GetComponent<Text>().text = "Insert Note";
923  mNewNoteButton.onClick.RemoveListener( OnModifyNote );
924  mNewNoteButton.onClick.AddListener( OnCreateNote );
925 
926  // Reset the variables related to editing a note.
927  mEditPanel = null;
928  mEditing = false;
929 
930  // Reset the pitch selections.
931  mPitchSelector.ResetPitches();
932  mDrumSelector.ResetPitches();
933  }
934  }
