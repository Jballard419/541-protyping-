181  BlackKey key = null;
182

186  mKeyObjects[i].name = Music.NoteToString( mRepresentedPitches[i] ) + "Key";
187 
188  // Get the black key.
189  key = mKeyObjects[i].GetComponent<BlackKey>();
190  Assert.IsNotNull( key, "Could not get a reference to the BlackKey script!" );
191

193  key.Pitch = mRepresentedPitches[i];
194 
195  // Add listeners to its events.
196  key.BlackKeyPressed.AddListener( HandleBlackKeyPressed );

200  SpringJoint spring = mKeyObjects[i].GetComponent<SpringJoint>();
201  Assert.IsNotNull( spring, "Could not get spring joint of black key!" );
202 
203  // Get the offset position.

206

208  spring.autoConfigureConnectedAnchor = false;
209  spring.connectedAnchor += keyPosition;
210 
211  // Set the new position.
212  keyPosition.y = mKeyObjects[i].transform.position.y;
213  keyPosition.x = mKeyObjects[i].transform.position.x;
214  mKeyObjects[i].transform.position = keyPosition;
215 
216  // Put settings back on the spring joint.
217  spring.autoConfigureConnectedAnchor = true;
218 
219  // Set its place in the hierarchy.
220  mKeyObjects[i].transform.SetParent( transform, true );

392  /**
393  * @brief Handler for when a song begins playing.
394  * @param[in] aSong The song that began playing.
395  */
396  private void HandlePlaySongEvent( Song aSong )
397  {
398  // Keep track of the current key layout.
399  Music.PITCH[] currentLayout = mRepresentedPitches;
400 
401  // Get the highest note in the song.
402  Music.PITCH highestPitch = aSong.GetHighestPitch();
403 
404  // Get the lowest note in the song.
405  Music.PITCH lowestPitch = aSong.GetLowestPitch();
406 
407  // Get the song data.
408  Song.CombinedNoteData[] noteData = aSong.GetNoteData();
409 
410  // Clear the keyboard.
411  ClearKeyboard();
412 
413  // Set the range of keys.
414  mRepresentedPitches = null;
415  int numPitches = (int)highestPitch - (int)lowestPitch + 1;
416  mRepresentedPitches = new Music.PITCH[numPitches];
417  mNumWhiteKeys = 0;
418  for( int i = 0; i < numPitches; i++ )
419  {
420  mRepresentedPitches[i] = (Music.PITCH)( (int)lowestPitch + i );
421  if( !Music.IsPitchABlackKey( mRepresentedPitches[i] ) )
422  {
423  mNumWhiteKeys++;
424  }
425  }
426 
427  // Load the proper keys.
428  LoadKeys();
429 
430  // Get the sample interval (seconds between waveform samples.)
431  float sampInt = VirtualInstrument.SAMPLE_INTERVAL;
432 
433  // Draw the lesson markers for each pitch.
434  foreach( Song.CombinedNoteData note in noteData )
435  {
436  if( note.MelodyData.Pitches != null )
437  {
438  foreach( Music.PITCH pitch in note.MelodyData.Pitches )
439  {
440  StartCoroutine( DrawLessonMarker( (int)pitch - (int)lowestPitch, (float)note.TotalOffset * sampInt, ( note.MelodyData.NumSamples * sampInt ) - .02f ) );
441  }
442  }
443  }
444 
445  }
