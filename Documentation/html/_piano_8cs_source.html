79  /**

81  * The Piano samples are available in three separate @link DefBID dynamics@endlink :
82  * @n pp: pianissimo (Very Soft) for @link DefVel velocities@endlink from 0 to 50.
83  * @n mf: mezzo-forte (Half-loud) for @link DefVel velocities@endlink from 51 to 75.
84  * @n ff: fortissimo (Very Loud) for @link DefVel velocities@endlink from 76 to 100.
85  */
86  protected override void InitializeBuiltInDynamics()
87  {
88  // Set the number of Built-In Dynamics.
89  mNumBuiltInDynamics = 3;
90 
91  // Set up the strings for creating the filenames.
92  mBuiltInDynamics = new string[mNumBuiltInDynamics];
93  mBuiltInDynamics[0] = "pp";
94  mBuiltInDynamics[1] = "mf";
95  mBuiltInDynamics[2] = "ff";
96  mBuiltInDynamicsThresholds = new int[mNumBuiltInDynamics];
97 
98  // Velocities between 0 & 50 will play the pianissimo samples.
99  // Velocities between 51 & 75 will play the mezzoforte samples.
100  // Velocities between 75 & 100 will play the fortissimo samples.
101  mBuiltInDynamicsThresholds[0] = 50;
102  mBuiltInDynamicsThresholds[1] = 75;
103  mBuiltInDynamicsThresholds[2] = 100;
104 
105  // Allocate the audio data.
106  mAudioData = new float[mNumBuiltInDynamics][][];
107  for( int i = 0; i < mNumBuiltInDynamics; i++ )
108  {
109  mAudioData[i] = new float[Music.MAX_SUPPORTED_NOTES][];
110  }
111 
112  }

114  /**
115  * @brief Creates the filenames from which to load the wav files.

117  * An example filename would be:
118  * @n "Resources/Audio/VirtualInstrument/Piano/Samples/ff/C4ff"
119  */
120  protected override void CreateFilenames()
121  {
122  // Set the base filepath and number of files.
123  mFilepath = "Audio/VirtualInstrument/Piano/Samples/";
124  mNumFiles = 258;
125 
126  // Initialize the array of filenames.
127  mFilenames = new string[mNumFiles];
128 
129  // Iterate through each dynamics value and create filenames for each supported note.
130  int index = 0;
131  for( int i = 0; i < mNumBuiltInDynamics; i++ )
132  {
133  for( int j = (int)mLowestSupportedPitch; j <= (int)mHighestSupportedPitch; j++ )
134  {
135  mFilenames[index] = mFilepath + mBuiltInDynamics[i] + "/" + Music.NoteToString( j ) + mBuiltInDynamics[i];
136  index++;
137  }
138  }
