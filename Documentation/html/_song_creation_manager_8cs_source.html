327  /**
328  * @brief Creates a new note in the song.
329  *
330  * Gets the currently selected values from the modules in the scene uses them
331  * to @link Music::CreateNote create@endlink a new @link Music::CombinedNote note@endlink.
332  *
333  * The function won't do anything if no @link SongCreationManager::mPitchSelector pitches@endlink
334  * or @link SongCreationManager::mDrumSelector drums@endlink are selected.
335  *
336  * @see SongCreationManager::mNewNoteButton
337  */
338  public void OnCreateNote()
339  {
340  // Sanity check.
341  if( mEditing )
342  {
343  Assert.IsFalse( !mEditing, "Tried to create a note in the song creation interface when we should have been editing one!" );
344  return;
345  }
346 
347  // Get the selected pitches
348  Music.PITCH[] pitches = mPitchSelector.GetSelectedPitches();
349 
350  // Get the selected drums.
351  Music.PITCH[] drums = mDrumSelector.GetSelectedPitches();
352 
353  // Make sure that some pitches or drums are actually selected.
354  if( pitches != null || drums != null )
355  {
356  // Get the offset and velocity.
357  Music.NOTE_LENGTH length = Music.NOTE_LENGTH.NONE;
358  mLastLength = mLengthPanel.GetSelected();
359  Music.NOTE_LENGTH offset = mOffsetPanel.GetSelected();
360  int pitchVelocity = 0;
361  int drumVelocity = 0;
362 
363  // If the song doesn't have any notes, then set the offset of the first note to none.
364  if( mSong.GetNumNotes() == 0 )
365  {
366  offset = Music.NOTE_LENGTH.NONE;
367  }
368 
369  // Convert the selected pitches from the drum selector into actual drum hits.
370  Music.DRUM[] Hits = null;
371  if( drums != null )
372  {
373  Hits = new Music.DRUM[drums.Length];
374  for( int i = 0; i < drums.Length; i++ )
375  {
376  Hits[i] = (Music.DRUM)drums[i];
377  }
378  }
379 
380 
381  // Update the velocity/length for the pitches and drums if needed.
382  if( pitches != null )
383  {
384  pitchVelocity = (int)mVelocitySlider.value;
385  length = mLengthPanel.GetSelected();
386  }
387 
388  if( drums != null )
389  {
390  drumVelocity = (int)mVelocitySlider.value;
391  }
392 
393 
394  // Add the note to the song.
395  mSong.AddNote( Music.CreateNote( pitchVelocity, length, pitches, drumVelocity, Hits, offset ) );
396 
397  // Add the note to the note display.
398  mNoteDisplay.AddNote( pitchVelocity, length, pitches, drumVelocity, Hits, offset );
399 
400  // Reset the pitches.
401  mPitchSelector.ResetPitches();
402  mDrumSelector.ResetPitches();
403 
404  // Update the selection of the offset panel since most cases will use the previous length.
405  mOffsetPanel.SetSelected( mLengthPanel.GetSelected() );
406  }
407  }

409  /**

421  if( mEditPanel != null )
422  {
423  // Stop editing the current note/note panel.
424  mEditPanel.StopEditing();
425 
426  // Change the note/note panel being edited to the new one.
427  mEditPanel = aPanel;
428  }
429  // Handle when we start editing a panel.
430  else
431  {
432  // Update the new note button to be a modify note button.

447

449  Music.PITCH[] drumsAsPitches = null;

457  index++;
458  }
459  }
460  mDrumSelector.SetPitches( drumsAsPitches );
461  mLengthPanel.SetSelected( noteBeingModified.MusicalNote.Length );
462  mOffsetPanel.SetSelected( noteBeingModified.OffsetFromPrevNote );
463  mVelocitySlider.value = Mathf.Max( noteBeingModified.Drums.Velocity, noteBeingModified.MusicalNote.Velocity );
464  }
465  // Handle the case where we're editing a note/note panel and it calls this
466  // event while it's being edited. This means that the edit should be cancelled.
467  else
468  {
469  // Change the new note button back to its original state.
470  mNewNoteButton.transform.GetChild( 0 ).GetComponent<Text>().text = "Insert Note";
471  mNewNoteButton.onClick.RemoveListener( OnModifyNote );
472  mNewNoteButton.onClick.AddListener( OnCreateNote );
473 
474  // Mark that the panel is no longer being edited.
475  mEditPanel.StopEditing();
476 
477  // Reset the variables related to editing a note.
478  mLengthPanel.SetSelected( Music.NOTE_LENGTH.Q );
479  mOffsetPanel.SetSelected( Music.NOTE_LENGTH.NONE );

481  mPitchSelector.ResetPitches();
482  mDrumSelector.ResetPitches();

530  /**
531  * @brief Modifies a @link Music::CombinedNote note@endlink in the Song.
532  *
533  * @see Song::ReplaceNote
534  */
535  public void OnModifyNote()
536  {
537  // Make sure that we're actually editing a note.
538  if( mEditing )
539  {
540  // Sanity check.
541  if( mEditPanel == null )
542  {
543  Assert.IsNotNull( mEditPanel, "Tried to modify a note, but the note panel in the song creator was null!" );
544  return;
545  }
546 
547  // Get the pitches
548  Music.PITCH[] pitches = mPitchSelector.GetSelectedPitches();
549 
550  // Get the drums
551  Music.PITCH[] drums = mDrumSelector.GetSelectedPitches();
552 
553  // Make sure that some pitches or drums are actually selected.
554  if( pitches != null || drums != null )
555  {
556  // Get the index of the note in the song and the images for the length/offset.
557  int index = mEditPanel.GetNoteIndex();
558  Sprite[] images = mNoteDisplay.GetSprites();
559 
560  // Convert the selected pitches from the drum selector into actual drum hits.
561  Music.DRUM[] Hits = null;
562  if( drums != null )
563  {
564  Hits = new Music.DRUM[drums.Length];
565  for( int i = 0; i < drums.Length; i++ )
566  {
567  Hits[i] = (Music.DRUM)drums[i];
568  }
569  }
570 
571  // Set the values of the note struct.
572  Music.NOTE_LENGTH length = Music.NOTE_LENGTH.NONE;
573  Music.NOTE_LENGTH offset = mOffsetPanel.GetSelected();
574  int pitchVelocity = 0;
575  int drumVelocity = 0;
576 
577  // Update the velocity for the pitches and drums if needed.
578  if( pitches != null )
579  {
580  pitchVelocity = (int)mVelocitySlider.value;
581  length = mLengthPanel.GetSelected();
582  }
583 
584  if( drums != null )
585  {
586  drumVelocity = (int)mVelocitySlider.value;
587  }
588 
589  // Replace the note in the song with the modified note.
590  mSong.ReplaceNote( Music.CreateNote( pitchVelocity, length, pitches, drumVelocity, Hits, offset ), index );
591 
592  // Get the string of pitches for the modified note.
593  string pitchString = "None";
594  if( pitches != null )
595  {
596  pitchString = "";
597  for( int i = 0; i < pitches.Length; i++ )
598  {
599  if( pitches[i] == Music.PITCH.REST )
600  {
601  pitchString += "Rest ";
602  }
603  else
604  {
605  pitchString += ( Music.NoteToString( pitches[i] ) + " " );
606  }
607  }
608  }
609 
610  // Get the string of drums for the modified note.
611  string drumString = "None";
612  if( Hits != null )
613  {
614  drumString = "";
615  foreach( Music.DRUM drum in Hits )
616  {
617  drumString += Music.DrumToString( drum ) + " ";
618  }
619  }
620 
621  // Update the panel representing the note.
622  mEditPanel.SetDrums( drumString );
623  mEditPanel.SetPitches( pitchString );
624  mEditPanel.SetLengthImage( images[(int)length] );
625  mEditPanel.SetOffsetImage( images[(int)offset] );
626  mEditPanel.SetOffset( offset );
627  mEditPanel.SetVelocity( mVelocitySlider.value.ToString() );
628 
629  // Mark that the panel is no longer being edited.
630  mEditPanel.StopEditing();
631 
632  // Update the new note button so that it shows that we're no longer editing a note.
633  mNewNoteButton.transform.GetChild( 0 ).GetComponent<Text>().text = "Insert Note";
634  mNewNoteButton.onClick.RemoveListener( OnModifyNote );
635  mNewNoteButton.onClick.AddListener( OnCreateNote );
636 
637  // Reset the variables related to editing a note.
638  mEditPanel = null;
639  mEditing = false;
640 
641  // Reset the pitch selections.
642  mPitchSelector.ResetPitches();
643  mDrumSelector.ResetPitches();
644  }

684  if( mSong.GetSongType() == Song.SongType.DrumLoop )
