319  * @see @link DocSongFileFormat Song File Format@endlink Song::LoadSongFromFile
320  */
321  private void LoadSongs()
322  {
323  // Get the info for the files in the folder.
324  DirectoryInfo info = new DirectoryInfo( Song.SONG_FILE_PATH );
325 
326  // Iterate through each file in the folder and load a new song if the filename matches
327  // the format.
328  foreach( FileInfo file in info.GetFiles() )
329  {
330  if( file.Name.EndsWith( ".song" ) )
331  {
332  // Load the new song.
333  Song newSong = new Song( file.FullName );
334 
335  // Add the new song to the proper container.
336  if( file.Name.StartsWith( "SONG_" ) )
337  {
338  mCombinedSongs.Add( newSong.GetName(), newSong );
339  mNumCombinedSongs++;
340  }
341  else if( file.Name.StartsWith( "MELODY_" ) )
342  {
343  mMelodies.Add( newSong.GetName(), newSong );
344  mNumMelodies++;
345  }
346  else if( file.Name.StartsWith( "DRUMLOOP" ) )
347  {
348  mDrumLoops.Add( newSong.GetName(), newSong );
349  mNumDrumLoops++;
350  }
351  }
352  }
