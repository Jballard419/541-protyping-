81  public void AddNote( int aNoteIndex, int aMelodyVelocity, Music.NOTE_LENGTH aLength, Music.PITCH[] aPitches,
82  int aDrumVelocity, Music.DRUM[] aDrumHits, Music.NOTE_LENGTH aOffsetFromPrevNote )
83  {
84  // Declare variables to help create the note.
85  Sprite[] sprites = mParent.GetSprites();
86  string pitches = "None";
87  string drums = "None";
88 
89 
90  // Calculate the percentage used up in the measure.
91  float newPercentage = mPercentageUsed + Music.GetNoteLengthRelativeToMeasure( aOffsetFromPrevNote, Music.TIME_SIGNATURE_4_4() );
92 
93  // If there is no more room in the measure for this note, then send it back to the parent.
94  if( newPercentage >= 1f )
95  {
96  mParent.HandleFullMeasure( this, newPercentage - mPercentageUsed, aMelodyVelocity, aLength, aPitches, aDrumVelocity, aDrumHits, aOffsetFromPrevNote );
97  }
98 
99  // If there is more room, then create the new panel and add it to this measure.
100  else
101  {
102  // Update the percentage used.
103  mPercentageUsed = newPercentage;
104 
105  // Create a SC_NoteDisplayPanel
106  GameObject clone = null;
107  clone = Instantiate( Resources.Load<GameObject>( NOTE_DISPLAY_PANEL_PATH ) );
108  Assert.IsNotNull( clone, "Could not load SC_NoteDisplayPanel prefab!" );
109 
110  // Set the values for the note's transform.
111  clone.transform.SetParent( transform.parent );
112  clone.transform.localScale = Vector3.one;
113 
114  // Set the pitch string if needed.
115  if( aPitches != null )
116  {
117  pitches = "";
118  for( int i = 0; i < aPitches.Length; i++ )
119  {
120  if( aPitches[i] == Music.PITCH.REST )
121  {
122  pitches += "Rest ";
123  }
124  else
125  {
126  pitches += ( Music.NoteToString( aPitches[i] ) + " " );
127  }
128  }
129  }
130 
131  // Set the drum string if needed.
132  if( aDrumHits != null )
133  {
134  drums = "";
135  foreach( Music.DRUM drum in aDrumHits )
136  {
137  drums += Music.DrumToString( drum ) + " ";
138  }
139  }
140 
141  // Initialize the cloned panel.
142  SC_NoteDisplayPanel newPanel = clone.AddComponent<SC_NoteDisplayPanel>();
143  newPanel.SetDrums( drums );
144  newPanel.SetPitches( pitches );
145  newPanel.SetLengthImage( sprites[(int)aLength] );
146  newPanel.SetOffsetImage( sprites[(int)aOffsetFromPrevNote] );
147  newPanel.SetOffset( aOffsetFromPrevNote );
148  newPanel.SetVelocity( aMelodyVelocity.ToString() );
149  newPanel.SetParentContainer( this );
150  newPanel.SetNoteIndex( aNoteIndex );
151 
152  // Add the panel to the list.
153  mNotes.Add( newPanel );
154  }

228  /**
229  * @brief Removes a @link Music::CombinedNote note@endlink.
230  * @param[in] aNoteDisplayPanel The panel displaying the note to be removed.
231  */
232  public void OnRemoveNote( SC_NoteDisplayPanel aNoteDisplayPanel )
233  {
234  // If there is more than one note in the measure, then just remove it from the measure.
235  if( mNotes.Count > 1 )
236  {
237  // Find the note.
238  bool found = false;
239  int index = 0;
240  while( !found && index < mNotes.Count )
241  {
242  if( aNoteDisplayPanel == mNotes[index] )
243  {
244  found = true;
245  }
246  else
247  {
248  index++;
249  }
250  }
251  SC_NoteDisplayPanel temp = mNotes[index];
252 
253  // Remove the note from the list.
254  mNotes.RemoveAt( index );
255 
256  // Update the percentage used.
257  mPercentageUsed -= Music.GetNoteLengthRelativeToMeasure( temp.GetOffset(), Music.TIME_SIGNATURE_4_4() );
258 
259  // Destroy the note object.
260  DestroyImmediate( temp.gameObject, false );
261 
262  }
263  // If there aren't any notes left, then handle this measure being deleted.
264  else
265  {
266  SC_NoteDisplayPanel temp = mNotes[0];
267  mNotes.Clear();
268  DestroyImmediate( temp.gameObject, false );
269  mParent.HandleMeasureDeleted( this );
270  }
271  mParent.OnRemoveNote();
272  }
